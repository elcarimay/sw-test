#define MAX_NUMBER       200001
#define BUY             false
#define SELL            true
#define MAX_STOCK       6 // 1 - 5
#define INVALID_VALUE   0

struct Order {
    int stock;
    int quantity;
    int price;
    bool type;
};

Order orders[MAX_NUMBER];
int owp;

int minPrice[MAX_STOCK];
int maxProfit[MAX_STOCK];

void init() {
    owp = 0;
    for (int i = 1; i < MAX_NUMBER; i++)
        orders[i].quantity = 0;

    for (int i = 0; i < MAX_STOCK; i++)
    {
        minPrice[i] = 1000000;
        maxProfit[i] = -1000000;
    }
}

void conclude(int mBID, int mSID, bool type) {
    Order buy = orders[mBID];
    Order sell = orders[mSID];

    // 체결수량 갱신
    if (buy.quantity < sell.quantity) {
        sell.quantity -= buy.quantity;
        buy.quantity = 0;
    }
    else if (buy.quantity > sell.quantity) {
        buy.quantity -= sell.quantity;
        sell.quantity = 0;
    }
    else {  // buy.quantity == sell.quantity
        buy.quantity = 0;;
        sell.quantity = 0;
    }

    // 체결덮어 씌워주기
    orders[mBID] = buy;
    orders[mSID] = sell;

    // 체결 가격 정하기
    int price = -1;
    if (type == SELL) price = buy.price;
    else if (type == BUY) price = sell.price;

    // 최소값, 최대이익 매번 갱신
    int stock = orders[mBID].stock;
    if (price < minPrice[stock]) minPrice[stock] = price;
    if (price - minPrice[stock] > maxProfit[stock])
        maxProfit[stock] = price - minPrice[stock];

    return;
}

int getSellOrder(int mStock)
{
    /*
        50
            40
            30
            20
            10(3) 10(2) 10(4)
    */
    //1. 가장 낮은 가격?
    int minPrice = 2000000;
    for (int i = 1; i <= owp; i++)
    {
        //매도 주문인가?
        if (orders[i].type == BUY)    continue;
        if (orders[i].stock != mStock)    continue;

        //수량체크 필요
        if (orders[i].quantity == 0)  continue;

        if (orders[i].price < minPrice)  minPrice = orders[i].price;
    }

    if (minPrice == 2000000) return -1;

    // 2. '1.' 주문번호가 가장 낮은 주문?
    int idx = -1;
    int minNumber = 300000;
    for (int i = 1; i <= owp; i++)
    {
        //매도 주문인가?
        if (orders[i].type == BUY)    continue;
        if (orders[i].stock != mStock)    continue;

        // 수량체크 필요
        if (orders[i].quantity == 0)  continue;

        if (orders[i].price != minPrice)   continue;

        if (i < minNumber)
        {
            minNumber = i;
            idx = i;
        }
    }
    return idx;
}

int getBuyOrder(int mStock)
{
    //1. 가장 높은 가격?
    int maxPrice = 0;
    for (int i = 1; i <= owp; i++)
    {
        //매수 주문인가?
        if (orders[i].type == SELL)    continue;
        if (orders[i].stock != mStock)    continue;

        //수량체크 필요
        if (orders[i].quantity == 0)  continue;

        if (orders[i].price > maxPrice)  maxPrice = orders[i].price;
    }

    if (maxPrice == 0)   return -1;

    // 2. '1.' 주문번호가 가장 낮은 주문?
    int idx = -1;
    int minNumber = 300000;
    for (int i = 1; i < owp; i++)
    {
        //매수 주문인가?
        if (orders[i].type == SELL)    continue;
        if (orders[i].stock != mStock)    continue;

        // 수량체크 필요
        if (orders[i].quantity == 0)  continue;

        if (orders[i].price != maxPrice)   continue;

        if (i < minNumber)
        {
            minNumber = i;
            idx = i;
        }
    }
    return idx;
}

int buy(int mNumber, int mStock, int mQuantity, int mPrice) {
    // 매수 주문 생성
    orders[mNumber] = { mStock, mQuantity, mPrice, BUY };
    owp = mNumber;

    while (1)
    {
        if (orders[mNumber].quantity == 0)   break;

        // 체결이 가능한가?
        int sIdx = getSellOrder(mStock);
        if (sIdx == -1) break;

        // 찾은 매도 주문으로 체결이 가능한가?
        if (orders[sIdx].price > orders[mNumber].price) break;

        // 체결
        conclude(mNumber, sIdx, BUY);
    }
    // 현재 매수 주문의 남은 수량 반환
    return orders[mNumber].quantity;
}

int sell(int mNumber, int mStock, int mQuantity, int mPrice) {
    // 매도 주문 생성
    orders[mNumber] = { mStock, mQuantity, mPrice, SELL };
    owp = mNumber;

    // 체결 진행
    while (1)
    {
        // 현재 매도 주문으로 체결이 가능한 상태인가?
        if (orders[mNumber].quantity == 0)   break;

        // 매수 주문 찾기
        int bIdx = getBuyOrder(mStock);
        if (bIdx == -1) break;

        // 찾은 매수 주문으로 체결이 가능한가?
        if (orders[mNumber].price > orders[bIdx].price) break;

        // 체결
        conclude(bIdx, mNumber, SELL);
    }
    // 현재 매수 주문의 남은 수량 반환
    return orders[mNumber].quantity;
}

void cancel(int mNumber) {
    // 지금 들어온 주문번호가 어디 주문인지?
    orders[mNumber].quantity = INVALID_VALUE;
}

int bestProfit(int mStock) {
    return maxProfit[mStock];
}
