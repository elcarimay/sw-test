#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#define MAX 26
using namespace std;

int n;
int map[MAX][MAX];
int visit[MAX][MAX] = { 0 };
int dx[4] = { -1,0,1,0 };
int dy[4] = { 0,1,0,-1 };

//각 단지별 아파트 개수를 저장하기 위한 배열
vector<int> ans;
int cnt = 0;

int dfs(int x, int y) {
    visit[x][y] = 1;
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i], ny = y + dy[i];
        if (0 <= nx && nx < n && 0 <= ny && ny < n) {
            if (map[nx][ny] == 1 && visit[nx][ny] == 0) {
                dfs(nx, ny);
            }
        }
    }
    cnt++;
    return cnt;
}

struct Pos
{
    int x, y;
};


int bfs(int x, int y)
{
    visit[x][y] = 1;
    queue<Pos> Q;
    Q.push({ x, y });

    while (Q.empty() == false)
    {
        // Pos cur = Q.front();
        auto cur = Q.front(); Q.pop();

        for (int i = 0; i < 4; i++)
        {
            int nx = cur.x + dx[i], ny = cur.y + dy[i];
            if (0 <= nx && nx < n && 0 <= ny && ny < n) {
                if (map[nx][ny] == 1 && visit[nx][ny] == 0) {
                    cnt++;
                    visit[nx][ny] = 1;
                    Q.push({ nx, ny });
                }
            }
        }
    }
    return cnt;
}


int main() {
    cin >> n;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            scanf("%1d", &map[i][j]);

    int count = 0; //단지 수를 0으로 초기화 해준다.
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (map[i][j] == 1 && visit[i][j] == 0) {
                //ans.push_back(dfs(i, j));
                ans.push_back(bfs(i, j));
                count++;
                cnt = 0;
            }
        }
    }

    cout << count << endl;

    sort(ans.begin(), ans.end());
    for (auto m:ans)
    {
        cout << m << endl;
    }
    return 0;
}
