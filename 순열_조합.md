# 순열 ver. 1.1

```
def dfs(selected=[]):
    if len(selected) == n:
        result.append(selected[:]) ## deep copy
        return
    for i in range(len(a)):
        if a[i] not in selected:
            selected.append(a[i])
            dfs(selected)
            selected.pop()  ## Backtracking

a, n = [1, 2, 3], 2
result = []
dfs()
print(result)
```
# 조합 ver. 1.0
```
def dfs(j=0, selected=[]):
    if len(selected) == n:
        result.append(selected)
        return
    for i in range(j, len(a)):
        if a[i] not in selected:
            dfs(i, selected + [a[i]])

a, n = [1, 2, 3], 2
result = []
dfs()
print(result)
```
