#include <map>
#include <unordered_map>

#include <queue>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX = 201;

struct Data
{
	int dist;
	int row;
	int col;
};

struct Pos
{
	int y, x;
};

struct cmp
{
	const bool operator()(const Data& A, const Data& B) {
		if (A.dist == B.dist)
		{
			if (A.row == B.row)
				return B.col < A.col;

			return B.row < A.row;
		}
		return B.dist < A.dist;
	}
};

struct TimeEventData
{
	int time;
	int id;
};

struct cmpTime
{
	const bool operator()(const TimeEventData& A, const TimeEventData& B) {
		return B.time < A.time;
	}
};

struct MyBac
{
	vector<Pos> cellList;
	int deleted;
};

MyBac bacList[3'001];

priority_queue<TimeEventData, vector<TimeEventData>, cmpTime> eventTimeQ;


const int dx[] = { +1,-1 , +0,+0 };
const int dy[] = { +0,+0 , +1,-1 };

int N;
int ABS(int a) {
	if (a < 0)
		return -a;
	return a;
}

int getDist(int r1, int c1, int r2, int c2)
{
	int dist = ABS(r1 - r2);
	dist += ABS(c1 - c2);
	return dist;
}

int m[MAX][MAX];

int spaceNumber[MAX][MAX];	// row, col에 빈공간 id
vector<int> spaceList;		// count값

int visit[MAX][MAX];


bool timeEvetenHandler(int time)
{
	bool needRefresh = false;
	
	while (eventTimeQ.empty() == false && eventTimeQ.top().time <= time)
	{
		auto cur = eventTimeQ.top();
		eventTimeQ.pop();
		
		if (bacList[cur.id].deleted == true)
			continue;

		for (auto p : bacList[cur.id].cellList)
			m[p.y][p.x] = 0;

		bacList[cur.id].deleted = true;

		needRefresh = true;
	}

	return needRefresh;
}


void refreshSpace()
{
	for (int i = 1; i <= N; i++)
		for (int j = 1; j <= N; j++)
			spaceNumber[i][j] = -1;

	spaceList.clear();

	for (int i = 1; i <= N; i++)
	{
		for (int j = 1; j <= N; j++)
		{
			if (m[i][j] == 0 && spaceNumber[i][j] == -1)
			{
				queue<Pos> Q;
				Q.push({ i, j });

				spaceNumber[i][j] = spaceList.size();

				int cnt = 0;

				while (Q.empty() == false)
				{
					auto cur = Q.front(); Q.pop();

					cnt++;

					for (int d = 0; d < 4; d++)
					{
						int nextY = cur.y + dy[d];
						int nextX = cur.x + dx[d];

						// 범위 체크
						if (nextY < 1 || nextX < 1 || nextY > N || nextX > N)
							continue;

						// 이동 가능한지 여부
						if (m[nextY][nextX] != 0)
							continue;

						// 이미 체크한 장소
						if (spaceNumber[nextY][nextX] != -1)
							continue;


						spaceNumber[nextY][nextX] = spaceList.size();
						Q.push({ nextY, nextX });
					}
				}

				spaceList.push_back(cnt);
			}
		}
	}
}

void init(int _N)
{
	N = _N;

	for (int i = 1; i <= N; i++)
		for (int j = 1; j <= N; j++)
			m[i][j] = 0;

	refreshSpace();

	while (eventTimeQ.empty() == false)
		eventTimeQ.pop();

	for (int i = 1; i <= 3000; i++)
	{
		bacList[i].cellList.clear();
		bacList[i].deleted = false;
	}

}



Result putBacteria(int mTime, int mRow, int mCol, Bacteria mBac)
{
	bool needRefresh = timeEvetenHandler(mTime);
	
	Result ret = { 0, 0 };

	if (needRefresh)
		refreshSpace();

	// 이미 박테리아가 있는 경우
	if (m[mRow][mCol] != 0)
		return ret;

	// 빈공간이 없는 경우
	int spaceSize = spaceList[spaceNumber[mRow][mCol]];
	if(spaceSize < mBac.size)
		return ret;

	priority_queue<Data, vector<Data>, cmp> Q;
	for (int i = 1; i <= N; i++)
		for (int j = 1; j <= N; j++)
			visit[i][j] = 0;

	Q.push({ 0, mRow, mCol });

	visit[mRow][mCol] = 1;

	int size = mBac.size;

	vector<Pos> cellList;

	while (Q.empty() == false)
	{
		auto cur = Q.top(); Q.pop();

		m[cur.row][cur.col] = mBac.id;

		cellList.push_back({ cur.row, cur.col });

		size--;

		ret.row = cur.row;
		ret.col = cur.col;

		if (size == 0)
			break;

		for (int d = 0; d < 4; d++)
		{
			int nextY = cur.row + dy[d];
			int nextX = cur.col + dx[d];

			// 범위 체크
			if (nextY < 1 || nextX < 1 || nextY > N || nextX > N)
				continue;

			// 이동 가능한지 여부
			if (m[nextY][nextX] != 0)
				continue;

			// 이미 체크한 장소
			if (visit[nextY][nextX] != 0)
				continue;


			visit[nextY][nextX] = 1;

			int dist = getDist(mRow, mCol, nextY, nextX);
			Q.push({ dist, nextY, nextX });
		}
	}

	bacList[mBac.id].cellList = cellList;
	bacList[mBac.id].deleted = false;

	eventTimeQ.push({ mTime + mBac.time, mBac.id });

	refreshSpace();

	return ret;
}


int killBacteria(int mTime, int mRow, int mCol)
{
	bool needRefresh = timeEvetenHandler(mTime);
	
	int id = m[mRow][mCol];
	
	if (id == 0 && needRefresh) {
		refreshSpace();
		return 0;
	}

	if (id == 0)
		return 0;

	for (auto p : bacList[id].cellList)
		m[p.y][p.x] = 0;

	bacList[id].deleted = true;

	refreshSpace();

	return id;
}



int checkCell(int mTime, int mRow, int mCol)
{
	bool needRefresh = timeEvetenHandler(mTime);
	if (needRefresh)
		refreshSpace();

	return m[mRow][mCol];
}
